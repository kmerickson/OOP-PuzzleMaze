 74        super().__init__("Player", position, default_image)

 77        self.last_move_time = 0

 81        # print(f"[DEBUG] Drawing player at {self.rect.topleft}")

 84    # handle movement
 85    def update(self, maze: List[List[int]], doors: List[Door]) -> None:
 86        """Handles player input, basically copied handle_player_input()
 87        from previous version of game.py
 88        This version requires passing in the 'maze' which are the matrices
 89        created in the load_levels() function."""
 90
 91        # # print("[DEBUG] Player.update() CALLED")
 92        keys = pygame.key.get_pressed()
 93        # # print("Keys pressed:", [i for i, key in enumerate(keys) if key])
 94
 95        current_time = pygame.time.get_ticks()
 96        if current_time - self.last_move_time < PLAYER_MOVE_DELAY:
 97            return
 98
 99        keys = pygame.key.get_pressed()
100        dx = dy = 0
101        if keys[K_UP]:
102            dy = -1
103            # # print("Up pressed")
104        elif keys[K_DOWN]:
105            dy = 1
106            # # print("Down pressed")
107        elif keys[K_LEFT]:

109            # # print("Left pressed")

111            dx = 1

118    def _move(self, maze: List[List[int]], doors: List[Door], dy: int,
119              dx: int, current_time: int) -> None:
120        new_row = self.rect.top // TILE_SIZE + dy
121        new_col = self.rect.left // TILE_SIZE + dx
122        # print("Trying to  move to:", new_row, new_col)
123        if not (0 <= new_row < len(maze) and 0 <= new_col < len(maze[0])):
124            return
125
126        for door in doors:
127            if door.rect.collidepoint(new_col * TILE_SIZE, new_row * TILE_SIZE):
128                door.interact(self, maze)
129                tile_index = maze[new_row][new_col]
130                break
131
132        # utilize the maze matrix used to draw the level to
133        # decide what the player can do:
134        tile_index = maze[new_row][new_col]
135        # print("Tile at target:", tile_index)
136
137        if tile_index == TILE_KEY:
138            self.key_count += 1
139            print("Key Count: ", self.key_count)
140            maze[new_row][new_col] = TILE_EMPTY
141        elif tile_index == TILE_WALL:
142            return  # do nothing if wall
143        elif tile_index == TILE_DOOR:
144            return
145
146        # print(f"Moving player to: ({new_col * TILE_SIZE}, {new_row * TILE_SIZE})")
147        self.move_to(new_col * TILE_SIZE, new_row * TILE_SIZE)
148        self.last_move_time = current_time

151class Enemy(GameObject):
152    """Enemy GameObject class"""
153
154    def __init__(self, position: Tuple[int, int],
155                 image: pygame.Surface | None = None, velocity: int = 1) -> None:
156        if image is None:
157            default_image = pygame.image.load("assets/enemy.png").convert_alpha()
158            default_image = pygame.transform.scale(default_image, (TILE_SIZE, TILE_SIZE))
159        else:
160            default_image = image
161
162        super().__init__("Enemy", position, default_image)
163
164        self.velocity = velocity
165        self.last_move_time = 0
166
167    def update(self, maze: List[List[int]], player: GameObject) -> None:
168        """Update logic for the enemy class, pass in the maze 2D array and
169        the player game object."""
170        current_time = pygame.time.get_ticks()
171        if current_time - self.last_move_time < ENEMY_MOVE_DELAY:
172            return
173
174        new_row = self.rect.top // TILE_SIZE + self.velocity
175        col = self.rect.left // TILE_SIZE   # doesn't actually move from column
176
177        if 0 <= new_row < len(maze):
178            tile_index = maze[new_row][col]
179            if tile_index != TILE_WALL:
180                self.move_to(col * TILE_SIZE, new_row * TILE_SIZE)
181            else:
182                self.velocity *= -1
183        else:
184            self.velocity *= -1  # added to reverse if it goes out of bounds as well
185
186        self.last_move_time = current_time
187
188        if self.collides_with(player):
189            print("You Died!")
190            # ADDED to make return to main menu with player death
191            # Will return if player presses escape or dies, as an escape key click is mocked
192            mock_event = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_ESCAPE)
193            pygame.event.post(mock_event)

199class ObjectState(ABC, Generic[T]):
200    """Base abstract state class. Declares methods that all concrete states
201    should implement. Designed to take in any object class with state implementations"""
202
203    @property
204    def context(self) -> T:
205        """returns the game object which is this class's context"""
206        return self._context
207
208    @context.setter
209    def context(self, context: T) -> None:
210        """Setter function for the context, which is the object class"""
211        self._context = context

214    def handle(self, player: Player, maze: List[List[int]]) -> None:
215        """handle interactions with the player"""
216        pass
217
218    @abstractmethod
219    def update(self) -> None:
220        """Update variables"""
221        pass

226    by the ObjectState class. The two states are the LockedDoorState and

229    def __init__(self, position: Tuple[int, int], state: ObjectState['Door']) -> None:
230
231        default_image = pygame.image.load("assets/door.png").convert_alpha()
232        default_image = pygame.transform.scale(default_image, (TILE_SIZE, TILE_SIZE))
233
234        super().__init__("Door", position, default_image)
235
236        self._state: ObjectState['Door'] | None
237        self.transition_to(state)
238
239    def transition_to(self, state: ObjectState['Door']) -> None:
240        """Handles state transitions"""
241        # might be a better function to handle image transitions
242        # maybe by storing the default image in the state
243        self._state = state
244        self._state.context = self
245
246    def interact(self, player: Player, maze: List[List[int]]) -> None:
247        """Passes the player object in to state object to handle behavior.
248        Maze is passed in to update images."""
249        if self._state is not None:
250            self._state.handle(player, maze)
251
252    def update(self) -> None:
253        """Delegates updating to the state class."""
254        if self._state is not None:
255            self._state.update()

257    def draw(self, screen: pygame.Surface) -> None:
258        """Utilizes basic drawing operation from GameObject class"""
259        super().draw(screen)
